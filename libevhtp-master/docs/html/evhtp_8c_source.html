01331 static evbuf_t *
01332 _evhtp_create_reply(evhtp_request_t * request, evhtp_res code) {
01333     evbuf_t    * buf          = evbuffer_new();
01334     const char * content_type = evhtp_header_find(request->headers_out, "Content-Type");
01335     char         res_buf[1024];
01336     int          sres;
01337 
01338     if (htparser_get_multipart(request->conn->parser) == 1) {
01339         goto check_proto;
01340     }
01341 
01342     if (evbuffer_get_length(request->buffer_out) && request->chunked == 0) {
01343         /* add extra headers (like content-length/type) if not already present */
01344 
01345         if (!evhtp_header_find(request->headers_out, "Content-Length")) {
01346             char lstr[128];
01347 #ifndef WIN32
01348             sres = snprintf(lstr, sizeof(lstr), "%zu",
01349                             evbuffer_get_length(request->buffer_out));
01350 #else
01351             sres = snprintf(lstr, sizeof(lstr), "%u",
01352                             evbuffer_get_length(request->buffer_out));
01353 #endif
01354 
01355             if (sres >= sizeof(lstr) || sres < 0) {
01356                 /* overflow condition, this should never happen, but if it does,
01357                  * well lets just shut the connection down */
01358                 request->keepalive = 0;
01359                 goto check_proto;
01360             }
01361 
01362             evhtp_headers_add_header(request->headers_out,
01363                                      evhtp_header_new("Content-Length", lstr, 0, 1));
01364         }
01365 
01366         if (!content_type) {
01367             evhtp_headers_add_header(request->headers_out,
01368                                      evhtp_header_new("Content-Type", "text/plain", 0, 0));
01369         }
01370     } else {
01371         if (!evhtp_header_find(request->headers_out, "Content-Length")) {
01372             const char * chunked = evhtp_header_find(request->headers_out,
01373                                                      "transfer-encoding");
01374 
01375             if (!chunked || !strstr(chunked, "chunked")) {
01376                 evhtp_headers_add_header(request->headers_out,
01377                                          evhtp_header_new("Content-Length", "0", 0, 0));
01378             }
01379         }
01380     }
01381 
01382 check_proto:
01383     /* add the proper keep-alive type headers based on http version */
01384     switch (request->proto) {
01385         case EVHTP_PROTO_11:
01386             if (request->keepalive == 0) {
01387                 /* protocol is HTTP/1.1 but client wanted to close */
01388                 evhtp_headers_add_header(request->headers_out,
01389                                          evhtp_header_new("Connection", "close", 0, 0));
01390             }
01391             break;
01392         case EVHTP_PROTO_10:
01393             if (request->keepalive == 1) {
01394                 /* protocol is HTTP/1.0 and clients wants to keep established */
01395                 evhtp_headers_add_header(request->headers_out,
01396                                          evhtp_header_new("Connection", "keep-alive", 0, 0));
01397             }
01398             break;
01399         default:
01400             /* this sometimes happens when a response is made but paused before
01401              * the method has been parsed */
01402             htparser_set_major(request->conn->parser, 1);
01403             htparser_set_minor(request->conn->parser, 0);
01404             break;
01405     } /* switch */
01406 
01407 
01408     /* attempt to add the status line into a temporary buffer and then use
01409      * evbuffer_add(). Using plain old snprintf() will be faster than
01410      * evbuffer_add_printf(). If the snprintf() fails, which it rarely should,
01411      * we fallback to using evbuffer_add_printf().
01412      */
01413 
01414     sres = snprintf(res_buf, sizeof(res_buf), "HTTP/%d.%d %d %s\r\n",
01415                     htparser_get_major(request->conn->parser),
01416                     htparser_get_minor(request->conn->parser),
01417                     code, status_code_to_str(code));
01418 
01419     if (sres >= sizeof(res_buf) || sres < 0) {
01420         /* failed to fit the whole thing in the res_buf, so just fallback to
01421          * using evbuffer_add_printf().
01422          */

01463 static void
01464 _evhtp_connection_readcb(evbev_t * bev, void * arg) {
01465     evhtp_connection_t * c = arg;
01466     void               * buf;
01467     size_t               nread;
01468     size_t               avail;
01469 
01470     avail = evbuffer_get_length(bufferevent_get_input(bev));
01471 
01472     if (c->request) {
01473         c->request->status = EVHTP_RES_OK;
01474     }
01475 
01476     if (c->paused == 1) {
01477         return;
01478     }
01479 
01480     buf = evbuffer_pullup(bufferevent_get_input(bev), avail);
01481 
01482     bufferevent_disable(bev, EV_WRITE);
01483     {
01484         nread = htparser_run(c->parser, &request_psets, (const char *)buf, avail);
01485     }
01486     bufferevent_enable(bev, EV_WRITE);
01487 
01488     if (c->owner != 1) {
01489         /*
01490          * someone has taken the ownership of this connection, we still need to
01491          * drain the input buffer that had been read up to this point.
01492          */
01493         evbuffer_drain(bufferevent_get_input(bev), nread);
01494         evhtp_connection_free(c);
01495         return;
01496     }
01497 
01498     if (c->request) {
01499         switch (c->request->status) {
01500             case EVHTP_RES_DATA_TOO_LONG:
01501                 if (c->request->hooks && c->request->hooks->on_error) {
01502                     (*c->request->hooks->on_error)(c->request, -1, c->request->hooks->on_error_arg);
01503                 }
01504                 evhtp_connection_free(c);
01505                 return;
01506             default:
01507                 break;
01508         }
01509     }
01510 
01511     evbuffer_drain(bufferevent_get_input(bev), nread);
01512 
01513     if (c->request && c->request->status == EVHTP_RES_PAUSE) {
01514         evhtp_request_pause(c->request);
01515     } else if (avail != nread) {
01516         evhtp_connection_free(c);
01517     }
01518 } /* _evhtp_connection_readcb */

01520 static void
01521 _evhtp_connection_writecb(evbev_t * bev, void * arg) {
01522     evhtp_connection_t * c = arg;
01523 
01524     if (c->request == NULL) {
01525         return;
01526     }
01527 
01528     _evhtp_connection_write_hook(c);
01529 
01530     if (c->paused == 1) {
01531         return;
01532     }
01533 
01534     if (c->request->finished == 0 || evbuffer_get_length(bufferevent_get_output(bev))) {
01535         return;
01536     }
01537 
01538     /*
01539      * if there is a set maximum number of keepalive requests configured, check
01540      * to make sure we are not over it. If we have gone over the max we set the
01541      * keepalive bit to 0, thus closing the connection.
01542      */
01543     if (c->htp->max_keepalive_requests) {
01544         if (++c->num_requests >= c->htp->max_keepalive_requests) {
01545             c->request->keepalive = 0;
01546         }
01547     }
01548 
01549     if (c->request->keepalive) {
01550         _evhtp_request_free(c->request);
01551 
01552         c->request         = NULL;
01553         c->body_bytes_read = 0;
01554 
01555         if (c->htp->parent && c->vhost_via_sni == 0) {
01556             /* this request was servied by a virtual host evhtp_t structure
01557              * which was *NOT* found via SSL SNI lookup. In this case we want to
01558              * reset our connections evhtp_t structure back to the original so
01559              * that subsequent requests can have a different Host: header.
01560              */
01561             evhtp_t * orig_htp = c->htp->parent;
01562 
01563             c->htp = orig_htp;
01564         }
01565 
01566         htparser_init(c->parser, htp_type_request);
01567 
01568 
01569         htparser_set_userdata(c->parser, c);
01570         return;
01571     } else {
01572         evhtp_connection_free(c);
01573         return;
01574     }
01575 
01576     return;
01577 } /* _evhtp_connection_writecb */

02293 evhtp_unescape_string(unsigned char ** out, unsigned char * str, size_t str_len) {
02294     unsigned char    * optr;
02295     unsigned char    * sptr;
02296     unsigned char      d;
02297     unsigned char      ch;
02298     unsigned char      c;
02299     size_t             i;
02300     enum unscape_state state;
02301 
02302     if (out == NULL || *out == NULL) {
02303         return -1;
02304     }
02305 
02306     state = unscape_state_start;
02307     optr  = *out;
02308     sptr  = str;
02309     d     = 0;
02310 
02311     for (i = 0; i < str_len; i++) {
02312         ch = *sptr++;
02313 
02314         switch (state) {
02315             case unscape_state_start:
02316                 if (ch == '%') {
02317                     state = unscape_state_hex1;
02318                     break;
02319                 }
02320 
02321                 *optr++ = ch;
02322 
02323                 break;
02324             case unscape_state_hex1:
02325                 if (ch >= '0' && ch <= '9') {
02326                     d     = (unsigned char)(ch - '0');
02327                     state = unscape_state_hex2;
02328                     break;
02329                 }
02330 
02331                 c = (unsigned char)(ch | 0x20);
02332 
02333                 if (c >= 'a' && c <= 'f') {
02334                     d     = (unsigned char)(c - 'a' + 10);
02335                     state = unscape_state_hex2;
02336                     break;
02337                 }
02338 
02339                 state   = unscape_state_start;
02340                 *optr++ = ch;
02341                 break;
02342             case unscape_state_hex2:
02343                 state   = unscape_state_start;
02344 
02345                 if (ch >= '0' && ch <= '9') {
02346                     ch      = (unsigned char)((d << 4) + ch - '0');
02347 
02348                     *optr++ = ch;
02349                     break;
02350                 }
02351 
02352                 c = (unsigned char)(ch | 0x20);
02353 
02354                 if (c >= 'a' && c <= 'f') {
02355                     ch      = (unsigned char)((d << 4) + c - 'a' + 10);
02356                     *optr++ = ch;
02357                     break;
02358                 }
02359 
02360                 break;
02361         } /* switch */

02367 evhtp_query_t *
02368 evhtp_parse_query(const char * query, size_t len) {
02369     evhtp_query_t    * query_args;
02370     query_parser_state state   = s_query_start;
02371     char             * key_buf = NULL;
02372     char             * val_buf = NULL;
02373     int                key_idx;
02374     int                val_idx;
02375     unsigned char      ch;
02376     size_t             i;
02377 
02378     query_args = evhtp_query_new();
02379 
02380     if (!(key_buf = malloc(len + 1))) {
02381         return NULL;
02382     }
02383 
02384     if (!(val_buf = malloc(len + 1))) {
02385         free(key_buf);
02386         return NULL;
02387     }
02388 
02389     key_idx = 0;
02390     val_idx = 0;
02391 
02392     for (i = 0; i < len; i++) {
02393         ch = query[i];
02394 
02395         if (key_idx >= len || val_idx >= len) {
02396             goto error;
02397         }
02398 
02399         switch (state) {
02400             case s_query_start:
02401                 memset(key_buf, 0, len);
02402                 memset(val_buf, 0, len);
02403 
02404                 key_idx = 0;
02405                 val_idx = 0;
02406 
02407                 switch (ch) {
02408                     case '?':
02409                         state = s_query_key;
02410                         break;
02411                     case '/':
02412                         state = s_query_question_mark;
02413                         break;
02414                     default:
02415                         state = s_query_key;
02416                         goto query_key;
02417                 }
02418 
02419                 break;
02420             case s_query_question_mark:
02421                 switch (ch) {
02422                     case '?':
02423                         state = s_query_key;
02424                         break;
02425                     case '/':
02426                         state = s_query_question_mark;
02427                         break;
02428                     default:
02429                         goto error;
02430                 }
02431                 break;
02432 query_key:
02433             case s_query_key:
02434                 switch (ch) {
02435                     case '=':
02436                         state = s_query_val;
02437                         break;
02438                     case '%':
02439                         key_buf[key_idx++] = ch;
02440                         key_buf[key_idx] = '\0';
02441                         state = s_query_key_hex_1;
02442                         break;
02443                     default:
02444                         key_buf[key_idx++] = ch;
02445                         key_buf[key_idx]   = '\0';
02446                         break;
02447                 }
02448                 break;
02449             case s_query_key_hex_1:
02450                 if (!evhtp_is_hex_query_char(ch)) {
02451                     /* not hex, so we treat as a normal key */
02452                     if ((key_idx + 2) >= len) {
02453                         /* we need to insert \%<ch>, but not enough space */
02454                         goto error;
02455                     }
02456 
02457                     key_buf[key_idx - 1] = '%';
02458                     key_buf[key_idx++]   = ch;
02459                     key_buf[key_idx]     = '\0';
02460                     state = s_query_key;
02461                     break;
02462                 }
02463 
02464                 key_buf[key_idx++] = ch;
02465                 key_buf[key_idx]   = '\0';
02466 
02467                 state = s_query_key_hex_2;
02468                 break;
02469             case s_query_key_hex_2:
02470                 if (!evhtp_is_hex_query_char(ch)) {
02471                     goto error;
02472                 }
02473 
02474                 key_buf[key_idx++] = ch;
02475                 key_buf[key_idx]   = '\0';
02476 
02477                 state = s_query_key;
02478                 break;
02479             case s_query_val:
02480                 switch (ch) {
02481                     case ';':
02482                     case '&':
02483                         evhtp_kvs_add_kv(query_args, evhtp_kv_new(key_buf, val_buf, 1, 1));
02484 
02485                         memset(key_buf, 0, len);
02486                         memset(val_buf, 0, len);
02487 
02488                         key_idx            = 0;
02489                         val_idx            = 0;
02490 
02491                         state              = s_query_key;
02492 
02493                         break;
02494                     case '%':
02495                         val_buf[val_idx++] = ch;
02496                         val_buf[val_idx]   = '\0';
02497 
02498                         state              = s_query_val_hex_1;
02499                         break;
02500                     default:
02501                         val_buf[val_idx++] = ch;
02502                         val_buf[val_idx]   = '\0';
02503 
02504                         break;
02505                 }     /* switch */
02506                 break;
02507             case s_query_val_hex_1:
02508                 if (!evhtp_is_hex_query_char(ch)) {
02509                     /* not really a hex val */
02510                     if ((val_idx + 2) >= len) {
02511                         /* we need to insert \%<ch>, but not enough space */
02512                         goto error;
02513                     }
02514 
02515 
02516                     val_buf[val_idx - 1] = '%';
02517                     val_buf[val_idx++]   = ch;
02518                     val_buf[val_idx]     = '\0';
02519 
02520                     state = s_query_val;
02521                     break;
02522                 }
02523 
02524                 val_buf[val_idx++] = ch;
02525                 val_buf[val_idx]   = '\0';
02526 
02527                 state = s_query_val_hex_2;
02528                 break;
02529             case s_query_val_hex_2:
02530                 if (!evhtp_is_hex_query_char(ch)) {
02531                     goto error;
02532                 }
02533 
02534                 val_buf[val_idx++] = ch;
02535                 val_buf[val_idx]   = '\0';
02536 
02537                 state = s_query_val;
02538                 break;
02539             default:
02540                 /* bad state */
02541                 goto error;
02542         }       /* switch */
02543     }
02544 
02545     if (key_idx && val_idx) {
02546         evhtp_kvs_add_kv(query_args, evhtp_kv_new(key_buf, val_buf, 1, 1));
02547     }
02548 
02549     free(key_buf);
02550     free(val_buf);
02551 
02552     return query_args;
02553 error:
02554     free(key_buf);
02555     free(val_buf);
02556 
02557     return NULL;
02558 }     /* evhtp_parse_query */

02618 void
02619 evhtp_send_reply_chunk_start(evhtp_request_t * request, evhtp_res code) {
02620     evhtp_header_t * content_len;
02621 
02622     if (evhtp_response_needs_body(code, request->method)) {
02623         content_len = evhtp_headers_find_header(request->headers_out, "Content-Length");
02624 
02625         switch (request->proto) {
02626             case EVHTP_PROTO_11:
02627 
02628                 /*
02629                  * prefer HTTP/1.1 chunked encoding to closing the connection;
02630                  * note RFC 2616 section 4.4 forbids it with Content-Length:
02631                  * and it's not necessary then anyway.
02632                  */
02633 
02634                 evhtp_kv_rm_and_free(request->headers_out, content_len);
02635                 request->chunked = 1;
02636                 break;
02637             case EVHTP_PROTO_10:
02638                 /*
02639                  * HTTP/1.0 can be chunked as long as the Content-Length header
02640                  * is set to 0
02641                  */

02643 
02644                 evhtp_headers_add_header(request->headers_out,
02645                                          evhtp_header_new("Content-Length", "0", 0, 0));

02659                                  evhtp_header_new("Transfer-Encoding", "chunked", 0, 0));

02666             char lstr[128];
